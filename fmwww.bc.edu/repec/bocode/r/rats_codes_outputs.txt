open data pedroni_ppp.xls
calendar(panelobs=246,m) 1973:6
data(format=xls,org=columns) 1//1973:06 20//1993:11 ae rf wpi cpi uswpi uscpi country entry year month
set logexrate    = log(ae)
set logratio     = log(cpi/uscpi)
set logwpi_ratio = log(wpi/uswpi)

source panelfm.src
source paneldols.src


*===== EXAMPLE 2: Model specifications =====

* with constant (default)
@panelfm(bvec=||0.0||,notdum,print=short,average=simple)
# logexrate logratio
@paneldols(bvec=||0.0||,notdum,print=short,average=simple)
# logexrate logratio

* model with trend
@panelfm(bvec=||0.0||,det=trend,notdum,print=short,average=simple)
# logexrate logratio
@paneldols(bvec=||0.0||,det=trend,notdum,print=short,average=simple)
# logexrate logratio

* with time dummies
@paneldols(bvec=||0.0||,tdum,print=short,average=simple)
# logexrate logratio
@panelfm(bvec=||0.0||,tdum,print=short,average=simple)
# logexrate logratio

* with time dummies and trend
@paneldols(bvec=||0.0||,tdum,det=trend,print=short,average=simple)
# logexrate logratio
@panelfm(bvec=||0.0||,tdum,det=trend,print=short,average=simple)
# logexrate logratio

* with different lag options
@paneldols(bvec=||0.0||,dlags=5,lags=5,det=trend,notdum,print=short,average=simple)
# logexrate logratio
@panelfm(bvec=||0.0||,lags=5,det=trend,notdum,print=short,average=simple)
# logexrate logratio

*===== EXAMPLE 5: Short-T multivariate (T=46) =====

do J=1,20
   set logexrate    1977:04+(J-1)*246 1993:11+(J-1)*246 = %na
   set logratio     1977:04+(J-1)*246 1993:11+(J-1)*246 = %na

*===== EXAMPLE 7: Saving residuals and fitted values =====
* Reload data because Example 5 overwrote with NAs

open data pedroni_ppp.xls
calendar(panelobs=246,m) 1973:6
data(format=xls,org=columns) 1//1973:06 20//1993:11 ae rf wpi cpi uswpi uscpi country entry year month
set logexrate = log(ae)
set logratio  = log(cpi/uscpi)

@panelfm(bvec=||0.0||,lags=5,notdum,print=short,average=simple)
# logexrate logratio
set ehat_fm = %resids
set yhat_fm = logexrate - %resids

@paneldols(bvec=||0.0||,lags=5,dlags=5,notdum,print=short,average=simple)
# logexrate logratio
set ehat_dols = %resids
set yhat_dols = logexrate - %resids

stats ehat_fm
stats yhat_fm
stats ehat_dols
stats yhat_dols
